## SECURITY (ASTRO 6 CSP API)

follow 2026 astro-security best practices

### CSP Configuration

```javascript
// astro.config.mjs
export default defineConfig({
  security: {
    csp: {
      algorithm: 'SHA-384',
      directives: [
        "default-src 'self'",
        "script-src 'self' 'unsafe-inline' https://plausible.io",
        "style-src 'self' 'unsafe-inline' https://fonts.bunny.net",
        "font-src 'self' https://fonts.bunny.net",
        "img-src 'self' https: ",
        "connect-src 'self' https://api.cms.com",
      ],
    },
  },
  experimental: {
    csp: true,
  },
});
```

### CMS Integration

```typescript
// src/loaders/cms-loader.ts
import type { CollectionLoader } from 'astro:content';

interface CMSConfig {
  apiKey: string;
  endpoint: string;
  baseUrl?: string;
}

export function cmsLoader(config: CMSConfig): CollectionLoader {
  return {
    name: 'sekoya-cms',
    load: async ({ store }) => {
      const response = await fetch(
        `${config.baseUrl || 'https://api.cms.com'}/${config.endpoint}`,
        {
          headers: {
            'Authorization': `Bearer ${config.apiKey}`,
          },
        }
      );
      
      if (!response.ok) {
        throw new Error(`CMS API error: ${response.status}`);
      }
      
      const data = await response.json();
      return data;
    },
  };
}
```
